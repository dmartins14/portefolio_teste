Análise de sequências de DNA.

Este conjunto de funções permite a análise de sequências de DNA através da
sua conversão para RNA, identificação de codões e deteção de open reading
frames (ORFs). As ORFs são procuradas nos seis frames de leitura possíveis,
considerando codões de início e de paragem. As ORFs identificadas podem
ser traduzidas em sequências de aminoácidos.

def dna_to_rna(dna: str) -> str:
    """
    Conversão de uma sequência de DNA em RNA.

    A função recebe uma string representando uma sequência de DNA,
    valida se contém apenas nucleotídeos válidos (A, C, T, G) e
    retorna a sequência correspondente de RNA, substituindo todas
    as timinas (T) por uracilos (U).
    """
    dna = dna.upper()

    for nt in dna:
        if nt not in "ACTG":
            raise ValueError("A sequência fornecida não é DNA válido.")

    rna = dna.replace("T", "U")
    return rna

def rna_reverse_complement(rna: str) -> str:
    """
    Sequência reversa complementar de uma sequência de RNA.

    A função recebe uma string representando uma sequência de RNA,
    gera a sequência complementar com base no complementaridade
    de bases (A↔U, C↔G) e inverte a ordem da sequência resultante.
    """
    transl = str.maketrans("AUCG", "UAGC")
    return rna.translate(transl)[::-1]

def get_codons(seq: str):
    """
    Divisão de uma sequência de nucleótidos em codões.

    A função percorre a sequência em grupos de três nucleótidos,
    começando no primeiro elemento da sequência, e retorna apenas
    os codões completos (que tenham exatamente três bases).
    """

    codoes = []

    for i in range(0, len(seq), 3):
        codao = seq[i:i+3]

        if len(codao) == 3:
            codoes.append(codao)

    return codoes

def frame_codons(seq: str, frame: int):
    """
    Divisão de uma sequência de nucleótidos em codões a partir de um frame específico.

    A função percorre a sequência em grupos de três nucleotídeos (codões),
    iniciando no índice indicado pelo frame, e retorna apenas os codões
    completos (com exatamente três bases).
    """
    codoes = []

    for i in range(frame, len(seq), 3):
        codao = seq[i:i+3]

        if len(codao) == 3:
            codoes.append(codao)

    return codoes




STOP = {"UAA", "UAG", "UGA"}
START = "AUG"

def find_orfs_in_frame(codoes):
    """
    Identificação de ORFs (Open Reading Frames) numa lista de codões.

    A função percorre uma lista de códons procurando um codão de
    início (START). Após encontrá-lo, acumula os codões seguintes até
    encontrar um codão de paragem (STOP), formando uma ORF completa.
    Cada ORF identificada é adicionada à lista de resultados.
    """
    orfs = []
    current = []
    active = False
    for cod in codoes:
        if cod == START:
            current = [cod]
            active = True
        elif active:
            current.append(cod)
            if cod in STOP:
                orfs.append(current.copy())
                active = False
    return orfs

def get_orfs_6frames(rna: str):
    """
    Identifica ORFs em todos os seis frames de leitura de uma sequência de RNA.

    A função analisa os três frames de leitura da sequência original de RNA
    e os três frames correspondentes da sequência complementar reversa.
    Para cada frame, os codões são extraídos e as ORFs (Open Reading Frames)
    são identificadas com base em codões de início e paragem.
    """
    rev = rna_reverse_complement(rna)
    frames = {}
    all_orfs = []

    for f in range(3):
        cds = frame_codons(rna, f)
        orfs = find_orfs_in_frame(cds)
        frames[f"+{f}"] = orfs
        all_orfs.extend(orfs)

    for f in range(3):
        cds = frame_codons(rev, f)
        orfs = find_orfs_in_frame(cds)
        frames[f"-{f}"] = orfs
        all_orfs.extend(orfs)

    return frames, all_orfs




GENETIC_CODE = {
    "UUU":"F","UUC":"F","UUA":"L","UUG":"L",
    "CUU":"L","CUC":"L","CUA":"L","CUG":"L",
    "AUU":"I","AUC":"I","AUA":"I","AUG":"M",
    "GUU":"V","GUC":"V","GUA":"V","GUG":"V",

    "UCU":"S","UCC":"S","UCA":"S","UCG":"S",
    "CCU":"P","CCC":"P","CCA":"P","CCG":"P",
    "ACU":"T","ACC":"T","ACA":"T","ACG":"T",
    "GCU":"A","GCC":"A","GCA":"A","GCG":"A",

    "UAU":"Y","UAC":"Y","UAA":"_","UAG":"_",
    "CAU":"H","CAC":"H","CAA":"Q","CAG":"Q",
    "AAU":"N","AAC":"N","AAA":"K","AAG":"K",
    "GAU":"D","GAC":"D","GAA":"E","GAG":"E",

    "UGU":"C","UGC":"C","UGA":"_","UGG":"W",
    "CGU":"R","CGC":"R","CGA":"R","CGG":"R",
    "AGU":"S","AGC":"S","AGA":"R","AGG":"R",
    "GGU":"G","GGC":"G","GGA":"G","GGG":"G"
}


def translate_orf(codoes):
    aminoacidos = []

    for c in codoes:
        if c not in GENETIC_CODE:
            raise ValueError(f"Codão inválido: {c}")

        aminoacidos.append(GENETIC_CODE[c])

    return "".join(aminoacidos)

def translate_all_orfs(orfs):
    """
    Tradução de todas as ORFs em sequências de aminoácidos.

    A função recebe uma lista de ORFs (Open Reading Frames), onde cada
    ORF é representada por uma lista de codões, e aplica a tradução de
    cada ORF para a sua sequência correspondente de aminoácidos.
    """
    return [translate_orf(orf) for orf in orfs]

def analyze_dna(dna: str):
    """
    Análise completa de uma sequência de DNA.

    Combina as funções anteriores, convertendo a sequência de 
    DNA em RNA, dividindo-a em codões, identificando ORFs 
    (Open Reading Frames) nos seis frames de leitura possíveis e 
    traduzindo todas as ORFs encontradas em sequências de aminoácidos.
    """
    rna = dna_to_rna(dna)
    codoes = get_codons(rna)
    frames, orfs = get_orfs_6frames(rna)
    proteinas = translate_all_orfs(orfs)

    return {
        "rna": rna,
        "codões": codoes,
        "Frames": frames,
        "ORFs": orfs,
        "Proteínas": proteinas,
        "Número de ORFs": len(orfs),
}
