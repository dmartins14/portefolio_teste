def dotplot(seq1, seq2):
    """
    Dotplot para comparação de sequências.

    Este método compara duas sequências carácter a carácter, construindo uma
    matriz binária onde é assinalada a coincidência entre caracteres nas mesmas
    posições relativas. O dotplot permite identificar regiões de similaridade
    e padrões conservados entre sequências.
    """
    matriz = []

    for i in range(len(seq1)):
        linha = []
        for j in range(len(seq2)):
            if seq1[i] == seq2[j]:
                linha.append(1)
            else:
                linha.append(0)
        matriz.append(linha)

    return matriz

def score_blosum(matriz, aa1, aa2):
    """
    Matriz de substituição e cálculo de score entre sequências.

    As matrizes de substituição permitem atribuir um score à substituição de
    um aminoácido por outro, refletindo a sua probabilidade evolutiva. Neste
    trabalho é utilizada a matriz BLOSUM62. Com base nesta matriz, é possível
    calcular o score total entre duas sequências de igual comprimento através
    da soma dos scores posição a posição.
    """
    if (aa1, aa2) in matriz:
        return matriz[(aa1, aa2)]
    else:
        return matriz[(aa2, aa1)]




def score_sequencias(seq1, seq2, matriz):
    score = 0

    for i in range(len(seq1)):
        score += score_blosum(matriz, seq1[i], seq2[i])

    return score


def nw_criarmatriz(seq1, seq2, gap):
    n = len(seq1)
    m = len(seq2)

    matriz = [[0] * (m + 1) for i in range(n + 1)]

    for i in range(1, n + 1):
        matriz[i][0] = i * gap

    for j in range(1, m + 1):
        matriz[0][j] = j * gap

    return matriz


def nw_preencher(matriz, seq1, seq2, match, mismatch, gap):
    for i in range(1, len(seq1) + 1):
        for j in range(1, len(seq2) + 1):
            if seq1[i - 1] == seq2[j - 1]:
                diagonal = matriz[i - 1][j - 1] + match
            else:
                diagonal = matriz[i - 1][j - 1] + mismatch

            cima = matriz[i - 1][j] + gap
            esquerda = matriz[i][j - 1] + gap

            matriz[i][j] = max(diagonal, cima, esquerda)

    return matriz


def nw_move(matriz, seq1, seq2, i, j, match, mismatch, gap):
    if i > 0 and j > 0:
        score = match if seq1[i - 1] == seq2[j - 1] else mismatch
        if matriz[i][j] == matriz[i - 1][j - 1] + score:
            return "diagonal"

    if i > 0 and matriz[i][j] == matriz[i - 1][j] + gap:
        return "cima"

    return "esquerda"


def nw_traceback(matriz, seq1, seq2, match, mismatch, gap):
    resultado1 = ""
    resultado2 = ""

    i = len(seq1)
    j = len(seq2)

    while i > 0 or j > 0:
        move = nw_move(matriz, seq1, seq2, i, j, match, mismatch, gap)

        if move == "diagonal":
            resultado1 = seq1[i - 1] + resultado1
            resultado2 = seq2[j - 1] + resultado2
            i -= 1
            j -= 1
        elif move == "cima":
            resultado1 = seq1[i - 1] + resultado1
            resultado2 = "-" + resultado2
            i -= 1
        else:
            resultado1 = "-" + resultado1
            resultado2 = seq2[j - 1] + resultado2
            j -= 1

    return resultado1, resultado2

def needleman_wunsch_dna(seq1, seq2, match=1, mismatch=-1, gap=-1):
    """
    Algoritmo de Needleman–Wunsch para alinhamento global.

    Este algoritmo realiza o alinhamento global de duas sequências utilizando
    programação dinâmica. Inicialmente é construída e preenchida uma matriz de
    scores com base em valores de match, mismatch e gap. De seguida, é efetuado
    um traceback que permite reconstruir o alinhamento ótimo, podendo introduzir
    gaps quando necessário.
    """
    if not seq1 or not seq2:
        raise ValueError("Sequências vazias")

    matriz = nw_criarmatriz(seq1, seq2, gap)
    matriz = nw_preencher(matriz, seq1, seq2, match, mismatch, gap)
    resultado1, resultado2 = nw_traceback(matriz, seq1, seq2, match, mismatch, gap)

    score = matriz[len(seq1)][len(seq2)]
    return resultado1, resultado2, score





def sw_criarmatriz(seq1, seq2):
    n = len(seq1)
    m = len(seq2)

    matriz = [[0] * (m + 1) for i in range(n + 1)]
    return matriz


def sw_preencher(matriz, seq1, seq2, match, mismatch, gap):
    for i in range(1, len(seq1) + 1):
        for j in range(1, len(seq2) + 1):
            if seq1[i - 1] == seq2[j - 1]:
                diag = matriz[i - 1][j - 1] + match
            else:
                diag = matriz[i - 1][j - 1] + mismatch

            cima = matriz[i - 1][j] + gap
            esquerda = matriz[i][j - 1] + gap

            matriz[i][j] = max(0, diag, cima, esquerda)

    return matriz


def sw_move(matriz, seq1, seq2, i, j, match, mismatch, gap):
    if matriz[i][j] == 0:
        return None

    if i > 0 and j > 0:
        score = match if seq1[i - 1] == seq2[j - 1] else mismatch
        if matriz[i][j] == matriz[i - 1][j - 1] + score:
            return "diagonal"

    if i > 0 and matriz[i][j] == matriz[i - 1][j] + gap:
        return "cima"

    return "esquerda"


def sw_traceback(matriz, seq1, seq2, match, mismatch, gap):
    resultado1 = ""
    resultado2 = ""

    max_score = 0
    i = 0
    j = 0

    for x in range(len(seq1) + 1):
        for y in range(len(seq2) + 1):
            if matriz[x][y] > max_score:
                max_score = matriz[x][y]
                i = x
                j = y

    while matriz[i][j] != 0:
        move = sw_move(matriz, seq1, seq2, i, j, match, mismatch, gap)

        if move == "diagonal":
            resultado1 = seq1[i - 1] + resultado1
            resultado2 = seq2[j - 1] + resultado2
            i -= 1
            j -= 1
        elif move == "cima":
            resultado1 = seq1[i - 1] + resultado1
            resultado2 = "-" + resultado2
            i -= 1
        else:
            resultado1 = "-" + resultado1
            resultado2 = seq2[j - 1] + resultado2
            j -= 1

    return resultado1, resultado2

def smith_waterman_dna(seq1, seq2, match=1, mismatch=-1, gap=-1):
    """
    Algoritmo de Smith–Waterman para alinhamento local.

    O algoritmo de Smith–Waterman permite identificar as regiões de maior
    similaridade entre duas sequências. Tal como o Needleman–Wunsch, utiliza
    programação dinâmica, no entanto, os scores negativos são substituídos
    por zero, ficando o alinhamento restrito às regiões localmente mais semelhantes.
    """
    matriz = sw_criarmatriz(seq1, seq2)
    matriz = sw_preencher(matriz, seq1, seq2, match, mismatch, gap)

    resultado1, resultado2 = sw_traceback(matriz, seq1, seq2, match, mismatch, gap)

    score = 0
    for linha in matriz:
        for valor in linha:
            if valor > score:
                score = valor

    return resultado1, resultado2, score




def inserir_gaps(alinhadas, nova_ref):
    resultado = []
    indice = 0

    for simbolo in nova_ref:
        if simbolo == "-":
            for i in range(len(alinhadas)):
                alinhadas[i] = alinhadas[i][:indice] + "-" + alinhadas[i][indice:]
            indice += 1
        else:
            indice += 1

    return alinhadas




def calcular_consenso(alinhamento):
    consenso = ""
    comprimento = len(alinhamento[0])

    for i in range(comprimento):
        contagem = {}
        for seq in alinhamento:
            caracter = seq[i]
            if caracter != "-":
                if caracter not in contagem:
                    contagem[caracter] = 1
                else:
                    contagem[caracter] += 1

        if len(contagem)>0:
            consenso += max(contagem, key=contagem.get)
        else:
            consenso += "-"

    return consenso


def inserir_gaps_alinhamento(alinhamento, referencia_antiga, referencia_nova):
    indice = 0

    for i in range(len(referencia_nova)):
        if referencia_nova[i] == "-" and referencia_antiga[i] != "-":
            for j in range(len(alinhamento)):alinhamento[j] = (alinhamento[j][:indice] + "-" + alinhamento[j][indice:])
            indice += 1
        else:
            indice += 1

    return alinhamento

def alinhamento_progressivo(seqs, match=1, mismatch=-1, gap=-1):
    """
    Alinhamento múltiplo progressivo.

    O alinhamento múltiplo progressivo é construído de forma incremental.
    Inicialmente, as duas primeiras sequências são alinhadas utilizando o
    algoritmo de Needleman–Wunsch. Em seguida, é calculada uma sequência de
    consenso, que é alinhada sucessivamente com cada nova sequência. Sempre
    que surgem gaps no consenso, estes são propagados a todas as sequências
    já alinhadas, garantindo a consistência do alinhamento.
    """
    resultado = needleman_wunsch_dna(seqs[0], seqs[1], match, mismatch, gap)
    alinhada1 = resultado[0]
    alinhada2 = resultado[1]

    alinhamento = [alinhada1, alinhada2]

    consenso = calcular_consenso(alinhamento)

    for i in range(2, len(seqs)):
        resultado = needleman_wunsch_dna(consenso, seqs[i], match, mismatch, gap)
        consenso_alinhado = resultado[0]
        nova_seq_alinhada = resultado[1]

        alinhamento = inserir_gaps_alinhamento(alinhamento, consenso, consenso_alinhado)

        alinhamento.append(nova_seq_alinhada)
        consenso = calcular_consenso(alinhamento)

    return alinhamento, consenso
